Si on obtient un certain rsultat et que nous voulions dcider quelle probabilit est la plus vraisemblable. On peut faire le rapport de vraisemblance

round(dbinom(0:10,10,.5)/dbinom(0:10,10,.7),digits=3)
 [1] 165.382  70.878  30.376  13.018   5.579   2.391   1.025   0.439   0.188   0.081   0.035

rapp_vrai1 <- round(dbinom(0:10,10,.7)/dbinom(0:10,10,.5),digits=3)
rapp_vrai0 <- round(dbinom(0:10,10,.5)/dbinom(0:10,10,.7),digits=3)
rapp_vrai <- rapp_vrai0/(rapp_vrai0+rapp_vrai1)
plot(0:10,
     rapp_vrai,
     type="h",
     lwd=10,
     col="magenta",
     xlab="Nombre de succs",
     ylab = "Probabilit de vraisemblance",
     ylim = c(0,1.1),
     ,main="Probabilit que p = 0.5"
     ,yaxt="n"
     ,xaxt="n",  # supprime l'axe X par dfaut
     )

# Ajouter laxe X avec ticks de 0  10
axis(side=1, at=0:10, labels=0:10)

# Axe des ordonnes (ticks de 0  1 par pas de 0.1)
axis(side=2, at=seq(0,1,0.25), labels=seq(0,1,0.25))

# Ajouter les valeurs au-dessus des btons
text(x = 0:10,
     y = rapp_vrai + 0.1,   # petit dcalage vers le haut
     labels = round(rapp_vrai,3),      # valeurs  afficher
     cex = 0.8,                        # taille du texte
     col = "blue")                     # couleur
